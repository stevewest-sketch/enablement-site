HTML Template for Training Card

Copy this code for each training card:

<!-- Training Card Template -->
<a href="#" class="training-card">
    <div class="training-header">
        <div class="training-date">[DATE]</div>
        <h3>[TRAINING TITLE]</h3>
        <span class="training-product">[PRODUCT NAME]</span>
    </div>
    <div class="training-body">
        <p>[DESCRIPTION]</p>
        <div class="training-meta">
            <span>[DURATION]</span>
            <span>[MATERIALS]</span>
        </div>
        <span class="training-link">View Materials →</span>
    </div>
</a>

To use in a grid layout:

<div class="content-grid">
    <!-- Add multiple training cards here -->
</div>

Anchor Fix Script for Google Sites

Add this script to every page with anchor navigation:

<script>
document.addEventListener('DOMContentLoaded', function() {
    // Update offset based on actual nav height
    function updateOffset() {
        const nav = document.querySelector('.quick-nav');
        if (nav) {
            const navHeight = nav.getBoundingClientRect().height + 16;
            document.documentElement.style.setProperty('--nav-offset', navHeight + 'px');
        }
    }
    
    // Handle anchor clicks
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function(e) {
            e.preventDefault();
            const targetId = this.getAttribute('href').substring(1);
            const targetElement = document.getElementById(targetId);
            
            if (targetElement) {
                targetElement.scrollIntoView({ 
                    behavior: 'smooth', 
                    block: 'start' 
                });
            }
        });
    });
    
    updateOffset();
    window.addEventListener('resize', updateOffset);
});
</script>
